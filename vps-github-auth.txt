    -e GITHUB_CLIENT_SECRET=placeholder_secret \
    -e GITHUB_REDIRECT_URI=http://100.85.227.83:8080/auth/github/callback \
    -e ALLOWED_USERS='["ubuntu","nana.gyasi"]' \
